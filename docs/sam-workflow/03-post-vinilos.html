📊 Comparación: GET vs POST

Aspecto GET /vinilos POST /vinilos
Propósito Leer datos Crear datos
Operación DynamoDB Scan PutItem

1. DynamoDBWritePolicy

SAM genera automáticamente una política IAM con permisos:
- dynamodb:PutItem
- dynamodb:UpdateItem
- dynamodb:DeleteItem

2. API Gateway Event

Events:
  CreateVinilo:
    Type: HttpApi
    Properties:
      Path: /vinilos      # Misma ruta que GET
      Method: post        # Diferente método
      ApiId: !Ref VinilosHttpApi  # Misma API

Importante: API Gateway enruta según método HTTP, no solo path.

  • GET /vinilos → GetVinilosFunction
  • POST /vinilos → CreateViniloFunction

Código: src/handlers/create-vinilo/index.js

const { DynamoDBClient } = require('@aws-sdk/client-dynamodb');
const { DynamoDBDocumentClient, PutCommand } = require('@aws-sdk/lib-dynamodb');
const { randomUUID } = require('crypto');

const ddbClient = new DynamoDBClient({ region: process.env.REGION });
const docClient = DynamoDBDocumentClient.from(ddbClient);

const TABLE_NAME = process.env.TABLE_NAME;

/**
 * Validate vinyl data
 */
function validateVinilo(vinilo) {
    const errors = [];

    // Required fields
    if (!vinilo.titulo || typeof vinilo.titulo !== 'string' || vinilo.titulo.trim().length === 0) {
        errors.push('titulo es requerido y debe ser un string no vacío');
    }

    if (!vinilo.artista || typeof vinilo.artista !== 'string' || vinilo.artista.trim().length === 0) {
        errors.push('artista es requerido y debe ser un string no vacío');
    }

    if (vinilo.precio === undefined || vinilo.precio === null) {
        errors.push('precio es requerido');
    } else if (typeof vinilo.precio !== 'number' || vinilo.precio <= 0) {
        errors.push('precio debe ser un número mayor que 0');
    }

    // Optional fields validation
    if (vinilo.año !== undefined && vinilo.año !== null) {
        if (typeof vinilo.año !== 'number' || vinilo.año < 1900 || vinilo.año > new Date().getFullYear()) {
            errors.push(`año debe ser un número entre 1900 y ${new Date().getFullYear()}`);
        }
    }

    return {
        valid: errors.length === 0,
        errors
    };
}

/**
 * Lambda handler
 */
exports.handler = async (event) => {
    console.log('Lambda invocada - POST vinilo');

    try {
        // 1. Parse request body
        let body;
        try {
            body = JSON.parse(event.body || '{}');
        } catch (parseError) {
            return {
                statusCode: 400,
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    error: 'Request body inválido',
                    mensaje: 'El body debe ser JSON válido'
                })
            };
        }

        // 2. Validate vinyl data
        const validation = validateVinilo(body);
        if (!validation.valid) {
            return {
                statusCode: 400,
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    error: 'Datos de vinilo inválidos',
                    errores: validation.errors
                })
            };
        }

        // 3. Generate ID if not provided
        const id = body.id || randomUUID();

        // 4. Build vinyl item for DynamoDB
        const vinilo = {
            id,
            titulo: body.titulo.trim(),
            artista: body.artista.trim(),
            precio: body.precio,
            año: body.año || null,
            imagen_key: body.imagen_key || 'placeholder.jpg',
            createdAt: new Date().toISOString(),
            updatedAt: new Date().toISOString()
        };

        // 5. Insert into DynamoDB (with duplicate prevention)
        const params = {
            TableName: TABLE_NAME,
            Item: vinilo,
            ConditionExpression: 'attribute_not_exists(id)' // Prevent duplicates
        };

        await docClient.send(new PutCommand(params));
        console.log('Vinilo created successfully:', id);

        // 6. Return success response
        return {
            statusCode: 201, // Created
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                mensaje: 'Vinilo creado exitosamente',
                vinilo: vinilo
            })
        };

    } catch (error) {
        console.error('Error:', error);

        // Handle duplicate ID (ConditionalCheckFailedException)
        if (error.name === 'ConditionalCheckFailedException') {
            return {
                statusCode: 409, // Conflict
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    error: 'Vinilo ya existe',
                    mensaje: 'Ya existe un vinilo con ese ID'
                })
            };
        }

        // Generic error
        return {
            statusCode: 500,
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                error: 'Error al crear vinilo',
                mensaje: error.message
            })
        };
    }
};

🔍 Análisis del Código

1. Códigos de Estado HTTP

Código Significado Cuándo usar
201 Created Recurso creado exitosamente
400 Bad Request Datos inválidos o JSON malformado
409 Conflict Recurso ya existe (duplicado)
500 Internal Server Error Error inesperado del servidor

2. randomUUID() - Generación de IDs

const { randomUUID } = require('crypto');

const id = body.id || randomUUID();
// Ejemplo: "f7d9c8a3-4b2e-4f1a-9d5c-8e7f6a5b4c3d"

✅ Ventajas:
- ✅ Único (colisiones prácticamente imposibles)
- ✅ No secuencial (no revela cantidad de items)
- ✅ Distribuido (funciona con múltiples regiones)

⚠️ Alternativas (NO usar en este caso):
- ❌ Math.random() - no criptográficamente seguro
- ❌ Incrementales (1, 2, 3...) - revelan información
- ❌ Timestamps - colisiones posibles

3. ConditionExpression - Prevenir Duplicados

const params = {
    TableName: TABLE_NAME,
    Item: vinilo,
    ConditionExpression: 'attribute_not_exists(id)'  // 👈 Importante!
};

await docClient.send(new PutCommand(params));

¿Qué hace?
- Si el item con ese id NO existe: Crea el item ✅
- Si el item con ese id YA existe: Lanza ConditionalCheckFailedException ❌

Sin ConditionExpression:

// ⚠️ PELIGRO: Sobrescribe item existente sin avisar
await docClient.send(new PutCommand(params));

4. Validación de Datos

function validateVinilo(vinilo) {
    const errors = [];

    if (!vinilo.titulo || typeof vinilo.titulo !== 'string' || vinilo.titulo.trim().length === 0) {
        errors.push('titulo es requerido y debe ser un string no vacío');
    }

    // ...más validaciones

    return {
        valid: errors.length === 0,
        errors
    };
}

✅ Buenas prácticas:
- ✅ Validar tipo de dato (typeof)
- ✅ Validar valor vacío (trim().length === 0)
- ✅ Devolver todos los errores, no solo el primero
- ✅ Mensajes de error descriptivos

🐛 Troubleshooting

Error: "Resource already exists in stack"

Causa: Ya desplegaste el recurso antes.

Solución: SAM detectará cambios automáticamente en siguiente deploy.

Error: "AccessDeniedException" al escribir en DynamoDB

Causa: Política IAM incorrecta.

Solución: Verifica que uses DynamoDBWritePolicy (no DynamoDBReadPolicy).

POST devuelve 500 pero GET funciona

Causa: Error en código Lambda (validación, parsing, etc.).

Solución: Ver logs:

sam logs -n CreateViniloFunction --stack-name {stack} --tail

Vinilos se duplican (no respeta ConditionExpression)

Causa: IDs diferentes generados en cada request.

Solución: Si quieres evitar duplicados por título+artista (no por ID), usa:

ConditionExpression: 'attribute_not_exists(titulo) AND attribute_not_exists(artista)'

⚠️ Esto requiere GSI (Global Secondary Index) - tema avanzado para más adelante.
